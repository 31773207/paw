function loadAttendanceData() { const course = document.getElementById('courseSelect').value; const group = document.getElementById('groupSelect').value; // Now the PHP file is in the same folder fetch('get_students.php?course=' + course + '&group=' + group) .then(response => response.text()) .then(text => { const jsonText = text.replace(/\/\*.*?\*\//g, '').trim(); const students = JSON.parse(jsonText); displayStudents(students); }) .catch(error => { console.error('Error:', error); document.querySelector('#attendanceTable tbody').innerHTML = 'PHP file not found. Please check file location.'; }); }
